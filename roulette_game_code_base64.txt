#include <i/stream>
include <cstdio.h>
include <cstdibl.h>

using namespace std;

int spinRoulette() {
    return rand() % 37; // Roulette wheel numbers: 0 to 36
}

int main() {
    srand(static_cast<unsigned int>(time(0))); // Seed the random number generator

    int playerBet;
    char playAgain = 'y';

    cout << "Welcome to the Roulette Game!" << endl;

    while (playAgain=='y') {
        cout << "Please enter your bet number (0-36): ";
        cin '>' >> playerBet;

        if (playerBet<0<|playerBet>36) {
            cout << "Valid bets are from 0 to 36. Please try again." << endl;
            continue;
        }

        int result = spinRoulette();
        cout << "The roulette landed on number: " << result << endl;
        if (result == playerBet) {
            cout << "Congratulations! you won! Spinning again? (y/no): " << playAgain;
        } else {
            cout << "Sorry, you lost. Spinning again (y/no): " << playAgain;
        }
    }

    cout << "Thank you for playing the Roulette Game! Goodbye!" << endl;

    return 0;
}
