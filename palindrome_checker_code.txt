Include <iostream>
#include <string>
#include <cctype>


// Function to check if a string is a palindrome
bool isPalindrome(const std::string& str) {
    // Initialize two pointers at the beginning and end of the string
    int left = 0;
    int right = str.length() - 1;

    // Traverse string from both ends towards the middle
    while (left < right) {
        // Convert characters to lower case to ensure case-insensitive comparison
        char leftChar = tolower(str[left]);
        char rightChar = tolower(str[right]);

        // Ignore non-alphanumeric characters
        if (!isalnum(leftChar)) {
            ++left;
        } else if (!isalnum(rightChar)) {
            --right;
        } else {
            // Compare characters
            if (leftChar != rightChar) {
                return false; // Not a palindrome
            }
            ++left;
            --right;
        }
    }
    return true; // It's a palindrome
}

int main() {
    std::string input;
    std::cout << "Enter a string: ";
    std::getline(std::cin, input);

    if (isPalindrome(input)) {
        std::cout << "The string is a palindrome.n";
    } else {
        std::cout << "The string is not a palindrome.n";
    }

    return 0;
}
