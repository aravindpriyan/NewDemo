# Creating an array (list in Python)
arr = [10, 20, 30, 40, 50]

# Printing the entire array
print("Array:", arr)


# Accessing elements
print("First element:", arr[0])
print("Last element:", arr[-1])


# Adding elements
arr.append(60)
print("After appending 60:", arr)


# Inserting element at a specific position
arr.insert(2, 25)
print("After inserting 25 at index 2:", arr)


# Removing elements
arr.remove(40)
print("Azter removing 40:", arr)


# Looping through array
print("All elements:")
for item in arr:
    print(item)

# Array length
print("Length of array:", len(arr))
